Extending

Because of the way that Blindsight is constructed, it is very easy to extend and adapt Blindsight to your needs.

Blindsight does this by breaking down the components of logging. In SLF4J, you have a logging API that looks like this: 

public interface Logger {
  boolean isInfoEnabled();
  void info(String message);
}

In Blindsight, the API and the logging levels are all traits, and the LoggerAPI is a composition of those traits. 

trait LoggerComponent[P, M] {
  type Predicate <: P
  type Method    <: M
  type Self
}

trait InfoLoggerAPI[P, M] extends LoggerComponent[P, M] {
  def isInfoEnabled: Predicate
  def info: Method
}

trait LoggerAPI[P, M]
    extends LoggerComponent[P, M]
    with TraceLoggerAPI[P, M]
    with DebugLoggerAPI[P, M]
    with InfoLoggerAPI[P, M]
    with WarnLoggerAPI[P, M]
    with ErrorLoggerAPI[P, M]

Here’s the SLF4J compatible API:

trait Logger extends LoggerAPI[LoggerPredicate, LoggerMethod]

trait LoggerPredicate {
  def apply(): Boolean
}

trait LoggerMethod {
  def apply(message: String): Unit
}

And here’s the fluent API:

trait FluentLogger extends LoggerAPI[LoggerPredicate, FluentLoggerMethod]

trait FluentAPI {
  def marker[T: ToMarkers](instance: => T): FluentLoggerMethod.Builder
  def message[T: ToMessage](instance: => T): FluentLoggerMethod.Builder
  def argument[T: ToArguments](instance: => T): FluentLoggerMethod.Builder
  def cause(e: Throwable): FluentLoggerMethod.Builder
}

trait FluentLoggerMethod extends FluentAPI {
  def apply[T: ToStatement](instance: => T): Unit
}

Breaking down the API means that you can pass through only the LoggerMethod, or assemble your custom logging levels. You have the option of extending LoggerMethod and LoggerPredicate with your own application specific logging API. Blindsight is designed to work with you so that adding new functionality is easy.

For example, if you want to add entry and exit logging levels to the SLF4J compatible API with automatically applied markers, you can do this:

import com.tersesystems.blindsight.api.{Arguments, ToArguments}
import com.tersesystems.blindsight.fluent.FluentLogger
import com.tersesystems.blindsight.logstash.ToArgumentsImplicits
import net.logstash.logback.argument.StructuredArguments
import org.slf4j.MarkerFactory
import sourcecode.{Args, Enclosing, File, Line}

import scala.util.{Failure, Success, Try}

class FlowLogger(logger: FluentLogger) {

  def flow[B]: FlowLoggerMethod[B] = new FlowLoggerMethod.Impl[B](logger, defaults[B])

  private def defaults[B]: ToArguments[B] = new ToArguments[B] {
    override def toArguments(instance: => B): Arguments = {
      instance match {
        case array: Array[_] =>
          Arguments(StructuredArguments.kv("result", array.toSeq))
        case other =>
          Arguments(StructuredArguments.kv("result", other))
      }
    }
  }

  def typedFlow[B: ToArguments]: FlowLoggerMethod[B] = {
    new FlowLoggerMethod.Impl[B](logger, implicitly[ToArguments[B]])
  }
}

trait FlowLoggerMethod[B] {

  def apply(block: => B)(implicit line: Line, file: File, enclosing: Enclosing, sourceArgs: Args): B

  def map(mappings: B => Arguments): FlowLoggerMethod[B]
}

object FlowLoggerMethod {

  class Impl[B](logger: FluentLogger, ta: ToArguments[B])
      extends FlowLoggerMethod[B]
      with ToArgumentsImplicits {
    private val entryLogger =
      logger.withMarker(MarkerFactory.getMarker("ENTRY"))
    private val exitLogger = logger.withMarker(MarkerFactory.getMarker("EXIT"))

    val markers = entryLogger.markers

    def map(mappings: B => Arguments): FlowLoggerMethod[B] = {
      new FlowLoggerMethod.Impl[B](logger, ToArguments(mappings))
    }

    override def apply(
        attempt: => B
    )(implicit line: Line, file: File, enclosing: Enclosing, sourceArgs: Args): B = {
      if (entryLogger.isTraceEnabled()) {
        entryLogger.trace
          .message(s"${enclosing.value} entry")
          .argument(sourceArgs)
          .logWithPlaceholders()
      }
      tryExecution(attempt)
    }

    protected def tryExecution(
        attempt: => B
    )(implicit line: Line, file: File, enclosing: Enclosing, sourceArgs: Args): B = {
      if (exitLogger.isTraceEnabled()) {
        val result = Try(attempt)
        result match {
          case Success(resultValue) =>
            val resultArgument: Arguments = ta.toArguments(resultValue)
            exitLogger.trace
              .message(s"${enclosing.value} exit")
              .argument(resultArgument)
              .logWithPlaceholders()
          case Failure(exception) =>
            exitLogger.trace
              .cause(exception)
              .message(s"${enclosing.value} exception")
              .logWithPlaceholders()
        }
        result.get
      } else {
        attempt
      }
    }
  }
}
